CREATE SEQUENCE cafe_seq
  START WITH 1
  INCREMENT BY 1
  NOMAXVALUE
  NOCYCLE
  NOCACHE;

CREATE TABLE cafe(
  ID NUMBER DEFAULT cafe_seq.NEXTVAL PRIMARY KEY,
  NAME VARCHAR2(50) NOT NULL,
  PRICE NUMBER NOT NULL
);

CREATE SEQUENCE my_blog
  START WITH 1
  INCREMENT BY 1
  MAXVALUE 999999999
  NOMAXVALUE
  MINVALUE 1
  NOMINVALUE
  CACHE 20
  NOCYCLE
  ORDER;
CREATE TABLE BLOG (
  id NUMBER DEFAULT my_blog.NEXTVAL PRIMARY KEY,
  name VARCHAR2(50),
  description VARCHAR2(200)
);
